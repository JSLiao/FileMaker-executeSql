executeSql 應用 (1) 產生 html table



透過 column separator, row separator 和 串接符號的操作和文字函數 讓 executeSql 輸出成 html 資料

如果要讓資料和 FileMaker 互動, 可以透過 javascript function

FileMaker.PerformScript, FileMaker.PerformScriptWithOption 把指定的資料傳送給 FileMaker

計算式:

let ( 
  [ 
    ~frame = "
      <!DOCTYPE html>
      <html lang='en'>
      <head>
        <meta charset='UTF-8'>
        <meta http-equiv='X-UA-Compatible' content='IE=edge'>
        <meta name='viewport' content='width=device-width, initial-scale=1.0'>
        <meta platform = 'FileMaker web viewer'>
        <title>executeSql to html content</title>
      </head>
      <body>
      <!--place ~tableContent here-->
      </body>
      </html>
    ";
    ~cS = "</td><td>" ;
    ~rS = "</td></tr><tr><td>" ;
    ~lead = "<table><tr><td>" ;
    ~trail = "</td></tr></table>";
    // 以下是計算
    ~tableContent = executeSql("
      Select name,addrCity,addrStr 
      From company 
      Where addrCity = '新北市' AND stroke > 16
      "; ~cS; ~rS );
    ~tableContent = ~lead & ~tableContent & ~trail
  ];
  substitute( ~frame ; "<!--place ~tableContent here-->"; ~tableContent )
)

/*
frame 如果使用雙引號 “ 要做 escape 處理
*/
let ( 
[ 
	// 輸入參數
	~fnString = "name,addrCity,addrStr,telAreaCode,telNo" ;
	~tn = "company" ;
	~cond1 = "addrCity";
	~cond2 = "stroke" ;
	~compare1 = "台中市" ;
	~compare2 = 1;
	~cSep = "</td><td>" ;
	~rSep = "</td></tr>¶<tr><td>" ;
	~lead = "<TABLE>¶<tr><td>" ;
	~trail = "</td></tr>¶</table>";
	// 進行計算
	~sql = executeSql ( "SELECT " & ~fnString & " FROM " & ~tn & " WHERE " & ~cond1 & " = ? AND " & ~cond2 & " > ? Fetch First 10 Rows Only" ; ~cSep ; ~rSep ; ~compare1 ; ~compare2 );
	~content = ~lead & ~sql & ~trail ;
	~htmlFrame = 
	"
	<!DOCTYPE html>
	<html lang=\"en\">
		<head>
			<meta charset=\"UTF-8\">
			<meta http-equiv=\"X-UA-Compatible\" content=\"IE=edge\">
			<meta name=\"viewport\" content=\"width=device-width, initial-scale=1.0\">
			<title>Document</title>
		</head>
		<body>
			<!--content here-->
		</body>
	</html>
	"
];
	substitute ( ~htmlFrame ; "<!--content here-->" ; ~content )
)

      "|| fieldId, " & ~sec3 &
      "|| fieldType, " & ~sec4 &
      "|| fieldClass, " & ~sec5 &
      "|| fieldReps " & ~sec6;
    ~content = ExecuteSQL (~query ; ~cs ; "" ; ~tName );
    ~content = ~title & ¶ & ~secLead & Substitute ( ~content ; ~cs ; "") & ~secTrail
  ];
  Substitute ( ~frame ; "<!--place content here-->" ; ~content )
  /*
  如果要顯示在 web Viewer
  "data:text/html," & Substitute ( ~frame ; "<!--place content here-->" ; ~content )
  */
