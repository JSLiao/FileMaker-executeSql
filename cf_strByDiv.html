cf_strByDiv



executeSql 不會對特殊字元做 escape 處理

當欄位的內容中包含 ¶ 或 , 如果使用內定的 column separator, row separator 則會造成取值錯誤

處理方式, 用特殊字元做為 column separator, row separator 

cf_strByDiv 依據 column separator, row separator 做取值, 取得正確的值

cf_strByDiv ( _text; _div; _ifDivCase; _nth )

version 2 updated

format:

    cf_strByDiv ( _text; _div; _ifDivCase; _nth; _ifNoDivReturn )

  parameters:

    _text: 用來取值的文字

    _div: 分隔字元或字串

    _ifDivCase: _div 是否區分大小寫

    _nth: 第幾個

    _ifNoDivReturn: true: 如果未發現 _div 是返回完整字串或, false: 返回""

計算式:

/*
  purpose:
    取處以 _div 作分隔的第 nth 個字串
  format:
    cf_strByDiv ( _text; _div; _ifDivCase; _nth )
  parameters:
    _text: 用來取值的文字
    _div: 分隔字元或字串
    _ifDivCase: _div 是否區分大小寫
    _nth: 第幾個
  dependencies:
    no
  returns:
    文字 或者 ""
    如果沒有分隔字元, 則 return empty
  recursive:
    no
  notes:
    從 cf_valueByDelimeter 修訂而來
  version current: v1
  by jeff liao
  initial:
    20220212
  history:
    
*/

case (
  _text = "" ; "" ;
  _div = "" ; "" ;
  filter ( _nth ; "0123456789") = _nth and not _nth > 0 ; "" ;
  let (
    [
      ~cS = "«" & char(41166) & "»" ;
      ~ifC = if ( _ifDivCase = true ; 1 ; 0 );
      ~txt = if ( ~ifC ; substitute ( _text ; _div ; ~cS ); _text );
      ~div = if ( ~ifC ; ~cs ; _div );
      ~lenDiv = length ( ~div );
      ~ifLeadDiv = if ( left ( ~txt ; ~lenDiv ) = ~div ; 1 ; 0 );
      ~ifTailDiv = if ( right ( ~txt ; ~lenDiv ) = ~div ; 1 ; 0 );
      ~max = patternCount ( ~txt ; ~div ) + 1 ;
      ~max = if ( ~ifLeadDiv ; ~max - 1 ; ~max );
      ~max = if ( ~ifTailDiv ; ~max - 1 ; ~max )
    ];
    case ( 
      patternCount ( ~txt ; ~div ) = 0 ; "" ; 
      _nth > ~max ; "" ;
      let ( 
        [
          ~txt = if ( ~ifLeadDiv ; ~txt ; ~div & ~txt );
          ~txt = if ( ~ifTailDiv ; ~txt ; ~txt & ~div );
          ~startPos = position ( ~txt ; ~div ; 1 ; _nth ) + ~lenDiv ;
          ~endPos = position ( ~txt ; ~div ; ~startPos ; 1 );
          ~len = ~endPos - ~startPos
        ];
        middle ( ~txt ; ~startPos ; ~len )
      )
    )
  )
)
